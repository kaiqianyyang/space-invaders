package invadem;

import processing.core.PApplet;
import processing.core.PImage;

public class Projectile {
    PImage fireImg;
    private int fireX;
    private int fireY;
    private boolean isFriendly;
    private boolean isRegular;
    public Projectile(int fireX,int fireY,boolean isFriendly, boolean isRegular){
        this.fireX = fireX;
        this.fireY = fireY;
        this.isFriendly = isFriendly;
        this.isRegular = isRegular;
    }

    public void setProjectileImg(PImage fireImg){
        this.fireImg = fireImg;
    }

    public boolean checkIfRegular(){
        return this.isRegular;
    }

    public PImage getProjectileImg(){
        return this.fireImg;
    }

    public void ProjectileMove(){
        this.fireY--;
    }

    public int getFireX(){
        return this.fireX;
    }

    public int getFireY(){
        return this.fireY;
    }

    public void unfriendlyMove(){
        this.fireY++;
    }

    public boolean isItFriently(){
        return isFriendly;
    }


}
