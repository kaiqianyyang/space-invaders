package invadem;

import processing.core.PApplet;
import processing.core.PImage;

public class Invader {
    protected PImage invaderImg;
    protected int invaderX;
    protected int invaderY;
    protected boolean hasBeenAttacked = false;
    private int hitPoint;
    public Invader(PImage invaderImg, int invaderX, int invaderY, int hitPoint){
        this.invaderImg = invaderImg;
        this.invaderX = invaderX;
        this.invaderY = invaderY;
        this.hitPoint = hitPoint;
    }

    public void changeSit(PImage open){
        this.invaderImg = open;
    }

    public PImage getInvaderImg(){
        return this.invaderImg;
    }

    public int getInvaderX(){
        return this.invaderX;
    }

    public int getInvaderY(){
        return this.invaderY;
    }

    public void invaderMoveLeft(){
        this.invaderX--;
    }

    public void invaderMoveRight(){
        this.invaderX++;
    }

    public void invaderMoveDown(){
        this.invaderY += 1;
    }

    public boolean isHasBeenAttacked(){
        return this.hasBeenAttacked;
    }

    public void invaderAttacked(){
        this.hasBeenAttacked = true;
    }

    public void invaderBlood(){
        this.hitPoint--;
    }

    public int checkHP(){
        return this.hitPoint;
    }

    public Projectile invaderFire(){
        Projectile projectile = new Projectile(this.invaderX+8,this.invaderY+16,false, true);
        return projectile;
    }

    public Projectile powerFire(){
        Projectile powerProjectile = new Projectile(this.invaderX+8,this.invaderY+16,false, false);
        return powerProjectile;
    }

}
