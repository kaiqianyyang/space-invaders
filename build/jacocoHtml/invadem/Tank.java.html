package invadem;
import processing.core.PImage;
import processing.core.PApplet;
public class Tank {
    private int setX;
    private int setY;
    private PImage tankImg;
    private int tankHP;
    private int superAmount;
    public Tank(PImage tankImg,int setX, int setY){
        this.setX = setX;
        this.setY = setY;
        this.tankImg = tankImg;
        this.tankHP = 3;
        this.superAmount = 5;
    }
    public int getSetX(){
        return this.setX;
    }

    public int getSetY(){
        return this.setY;
    }

    public PImage getPImage(){
        return this.tankImg;
    }

    public void moveLeft(){
        setX = setX - 1;
    }
    public void moveRight(){
        setX = setX + 1;
    }

    public Projectile tankFire(){
        Projectile projectile = new Projectile(this.setX+11,this.setY-5,true, true);
        return projectile;
    }
    //extension method
    public Projectile superTankFIre(){
        Projectile projectile = new Projectile(this.setX+11,this.setY-5,true, false);
        return projectile;
    }
    public void tankBlood(){
        tankHP --;
    }

    public void tankDead()
    {
        tankHP = 0;
    }
    public int getHP(){
        return this.tankHP;
    }

    public void newLevelTank(){
        this.tankHP = 3;
    }

    public void superFireRemains(){
        superAmount --;
    }

    public int getSuperAmount(){
        return this.superAmount;
    }
}
