package invadem;

import processing.core.PApplet;
import processing.core.PImage;
import java.util.ArrayList;
import java.util.List;

public class Barrier {

    private List<PImage> situation;
    private boolean isUseful = true;
    private int barrierX;
    private int barrierY;
    private PImage currentImg;

    public Barrier(List<PImage> situation ,int barrierX, int barrierY){
        this.barrierY = barrierY;
        this.barrierX = barrierX;
        this.situation = new ArrayList<>(situation);
        this.currentImg = situation.get(0);
    }

    public int getBarrierX(){
        return barrierX;
    }

    public int getBarrierY(){
        return barrierY;
    }

    public boolean useful(){
        return this.isUseful;
    }


    public void beAttacked(){
        if(this.situation.size() > 1){
            if(this.situation.size() == 2){
                this.isUseful = false;
            }
            this.situation.remove(0);
            this.currentImg = this.situation.get(0);
        }
    }

    public void PowerAttacked(){
        while(this.situation.size() > 1){
            this.beAttacked();
        }
    }

    public void refreshBarriers(){
        this.situation.clear();
        this.isUseful = true;
    }

    public PImage getCurrentImg() {
        return currentImg;
    }
}
